Factura Fecha Vendedor Cliente Productos Total Estado Acciones
{{ venta.numeroFactura }} {{ venta.fecha }} {{ venta.nombreVendedor || venta.idVendedor }} {{ venta.nombreCliente || venta.idCliente }} {{ linea.codigoProducto }} ({{ linea.cantidad }}) {{ formatoMoneda(venta.total) }} {{ venta.estado }}

📄 Detalle de Venta

Factura: {{ ventaDetalle.numeroFactura }}

Fecha: {{ ventaDetalle.fecha }}

Vendedor: {{ ventaDetalle.nombreVendedor || ventaDetalle.idVendedor }}

Cliente: {{ ventaDetalle.nombreCliente || ventaDetalle.idCliente }}

Estado: {{ ventaDetalle.estado }}

Productos:

Producto Cantidad Precio Unit. Descuento Subtotal
{{ linea.nombreProducto || linea.codigoProducto }} {{ linea.cantidad }} {{ formatoMoneda(linea.precioUnitario) }} {{ formatoMoneda(linea.descuento) }} {{ formatoMoneda(linea.subtotal) }}
Total: {{ formatoMoneda(ventaDetalle.total) }}
`, setup(props, { emit }) { const ventas = Vue.ref([]); const ventaDetalle = Vue.ref(null); const filtros = Vue.reactive({ estado: '', fecha: '' }); const cargarVentas = async () => { try { const response = await api.listarVentas(filtros); if (response.success) { ventas.value = response.data; } } catch (error) { emit('notificar', { mensaje: 'Error al cargar ventas: ' + error.message, tipo: 'error' }); } }; const aplicarFiltros = () => { cargarVentas(); }; const verDetalle = async (venta) => { try { const response = await api.obtenerVenta(venta.numeroFactura); if (response.success) { ventaDetalle.value = response.data; } } catch (error) { emit('notificar', { mensaje: 'Error al obtener detalle: ' + error.message, tipo: 'error' }); } }; const anularVenta = async (numeroFactura) => { if (!confirm('¿Está seguro de anular esta venta? Se revertirá el inventario.')) return; try { const response = await api.anularVenta(numeroFactura); if (response.success) { emit('notificar', { mensaje: 'Venta anulada correctamente', tipo: 'warning' }); await cargarVentas(); } } catch (error) { emit('notificar', { mensaje: 'Error al anular venta: ' + error.message, tipo: 'error' }); } }; const formatoMoneda = (valor) => { return new Intl.NumberFormat('es-GT', { style: 'currency', currency: 'GTQ' }).format(valor || 0); }; const getBadgeClass = (estado) => { const clases = { 'Completada': 'badge-success', 'Pendiente': 'badge-warning', 'Cancelada': 'badge-danger' }; return clases[estado] || 'badge-info'; }; Vue.onMounted(cargarVentas); return { ventas, ventaDetalle, filtros, aplicarFiltros, verDetalle, anularVenta, formatoMoneda, getBadgeClass }; } };
